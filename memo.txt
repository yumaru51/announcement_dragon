from django.contrib import admin
from models import model

class Model名Admin(admin.ModelAdmin):
    list_display = ('項目', '', ) # 画面上に表示
    list_filter = ['項目', '',  ] # 右側にフィルター表示
    search_fields = ['項目', ''] # 検索フィルターは一つだが、指定した項目で検索できる

①
from django.conf.urls import url
from django.urls import path
urlpatterns = [
    path('accounts/', include('django.contrib.auth.urls')),はログアウトとか
    path('admin/', admin.site.urls),は管理者画面とか
    path('url', view.関数名, name='name'),

urlpatterns = [
    path('url', view.def関数名, name='呼び出し名')
    path('url/<int:id>/<value>', view.def関数名, name='呼び出し名')※strは？
]
↓
def 関数名(request, id, value):※GETオンリーだよね？

template
推奨　<form action="{% url 'アプリ名:呼び出し名' %}" method="post">
非推奨　<form action="{% url '呼び出し名' %}" method="post">

パラメーター
<form action="{% url 'アプリ名:呼び出し名' パラメーター %}" method="post">
動的に
<form action="{% url 変数 %}" method="post">※変数を「''」で囲まない、パラメーターは？
